if not isfile("Overflow/key") or  api.check_key(readfile("Overflow/key")).code ~= "KEY_VALID" then
	local main = loadUI()
	local keyBox = main.KeyBox
	local getKey = main.GetKey
	local redeem = main.Redeem
	getKey.Activated:Connect(function()
		setclipboard("https://ads.luarmor.net/get_key?for=key-xizANDvIwBwq")
		getKey.Text = "Copied"
		wait(2)
		getKey.Text = "Get Key"
	end)
	redeem.Activated:Connect(function()
		if  api.check_key(keyBox.Text).code == "KEY_VALID" then
			redeem.Text = "Success"
			main.Parent:Destroy()
			writefile("Overflow/key", keyBox.Text)
			script_key = keyBox.Text
			api.load_script(); 
		else
			redeem.Text = "Check Output"
			warn(api.check_key(keyBox.Text).code)
		end
		wait(.5)
		if redeem then
			redeem.Text = "Redeem"
		end
	end)
elseif  isfile("Overflow/key") and  api.check_key(readfile("Overflow/key")).code== "KEY_VALID" then
	script_key = readfile("Overflow/key")
	api.load_script(); 
end
